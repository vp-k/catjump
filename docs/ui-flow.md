// src/utils/DeepLinkHandler.ts
class DeepLinkHandler {
  handleDeepLink(url: string) {
    const parsed = new URL(url);
    const path = parsed.pathname;

    switch (path) {
      case '/play':
        SceneManager.changeScene('GameScene');
        break;

      case '/shop':
        SceneManager.changeScene('ShopScene');
        break;

      case '/shop/cats':
        SceneManager.changeScene('ShopScene', { tab: 'cats' });
        break;

      case '/mission':
        SceneManager.changeScene('MissionScene');
        break;

      case '/leaderboard':
        SceneManager.changeScene('MenuScene');
        PopupManager.show('LeaderboardPopup');
        break;

      default:
        if (path.startsWith('/reward/')) {
          const rewardId = path.split('/')[2];
          SceneManager.changeScene('MenuScene');
          PopupManager.show('RewardPopup', { rewardId });
        }
        break;
    }
  }
}

// src/managers/SceneManager.ts
class SceneManager {
  private static currentScene: string;

  static changeScene(
    targetScene: string,
    data?: object,
    transition: 'fade' | 'slide' | 'none' = 'fade'
  ) {
    const game = Phaser.Game.instance;
    const currentScene = game.scene.getScene(this.currentScene);

    // ì „í™˜ ì• ë‹ˆë©”ì´ì…˜
    if (transition === 'fade') {
      currentScene.cameras.main.fadeOut(200);
      currentScene.time.delayedCall(200, () => {
        game.scene.start(targetScene, data);
        this.currentScene = targetScene;
      });
    } else {
      game.scene.start(targetScene, data);
      this.currentScene = targetScene;
    }
  }
}
